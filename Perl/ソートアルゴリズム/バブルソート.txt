# バブルソート

```clike
sub bubble_sort {
    my @nums = @_;;
    for ( my $x = 0 ; $x < @nums - 1 ; $x++ ) {
        for ( my $i = @nums - 1 ; $i > $x ; $i-- ) {
            if ( $nums[ $i - 1 ] > $nums[$i] ) {
                my $temp = $nums[ $i - 1 ];
                $nums[ $i - 1 ] = $nums[$i];
                $nums[$i] = $temp;
            }
        }
    }
    say Dumper \@nums;
}
```
